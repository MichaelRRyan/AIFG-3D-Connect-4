55int Minimax::minimax(GameBoard & t_board, Move t_move,
56 int t_depth, int t_alpha, int t_beta)
57{
58 // If the depth is odd, we're min.
59 bool isMin = t_depth % 2 == 1;
60
61 // Returns the max score if the AI will win this move or the min score if
62 // the enemy wins this turn.
63 if (FourTechEvaluator::isMoveAWin(t_board, t_move))
64 return (isMin) ? m_INT_MAX - 1 : m_INT_MIN + 1;
65
66 // If below the max depth or if ending on max (meaning t_move, which will
67 // will be evaluated, is the enemies move).
68 if (t_depth < m_maxDepth || !isMin)
69 {
70 // Performs the move we're checking.
71 t_board.setPiece(t_move.position, t_move.type);
72
73 // Calls the following lambda for each valid move.
74 forEachValidMove(t_board,
75 [&](Coordinate const& t_coord)
76 {
77 // Recursively calls minimax and takes the value returned.
78 int value = minimax(t_board,
79 { t_coord, opposite(t_move.type) },
80 t_depth + 1, t_alpha, t_beta);
81
82 // Sets the relevant value (alpha or beta) if applicable.
83 if (isMin) t_beta = std::min(t_beta, value);
84 else t_alpha = std::max(t_alpha, value);
85
86 // Should we prune?
87 if (t_alpha >= t_beta) return true; // Stops looping each move.
88 return false; // Continue looping each move.
89 });
90
91 // Returns the piece to it's original empty state and returns.
92 t_board.setPiece(t_move.position, PieceType::None);
93 return (isMin) ? t_beta : t_alpha; // Returns the relevant value.
94 }
95
96 // Return the best estimated coord if already met the depth.
97 return FourTechEvaluator::evaluateMove(t_board, t_move);
