27void Game::run()
28{
29 sf::Clock clock;
30 sf::Time timeSinceLastUpdate = sf::Time::Zero;
31 const float fps{ 60.0f };
32 sf::Time timePerFrame = sf::seconds(1.0f / fps); // 60 fps
33 while (!m_exitGame)
34 {
35 processEvents(); // as many as possible
36 timeSinceLastUpdate += clock.restart();
37 render(); // as many as possible
38 while (timeSinceLastUpdate > timePerFrame)
39 {
40 timeSinceLastUpdate -= timePerFrame;
41 processEvents(); // at least 60 fps
42 update(timePerFrame.asSeconds()); //60 fps
43 render(); // as many as possible
44 if (m_exitGame) break;
45 }
46 }
47}
